54  /// Verify that the 2 points found in Q are similar to 2 of the points in the base.
55  /// A filter by point feature : normal, distance, translation distance, angle and color.
56  /// Return a pair of bool, according of the right addition of the pair (p,q) or (q,p) in the congruent set.
57  template <typename WantedOptionsAndMore>
58  inline std::pair<bool,bool> operator() (const Point3D& p,
59  const Point3D& q,
60  typename Point3D::Scalar pair_normals_angle,
61  const Point3D& b0,
62  const Point3D& b1,
63  const WantedOptionsAndMore &options) {
64  static_assert( WantedOptionsAndMore::IS_ADAPTIVEPOINTFILTER_OPTIONS,
65  "Options passed to AdaptivePointFilter must inherit AdaptivePointFilter::Options" );
66  using Scalar = typename Point3D::Scalar;
67  using PairsVector = std::vector< std::pair<int, int> >;
68  using VectorType = typename Point3D::VectorType;
69 
70 
71  std::pair<bool,bool> res;
72  res.first = false;
73  res.second = false;
74 
75  VectorType segment1 = (b1.pos() - b0.pos()).normalized();
76 
77  if ( options.max_normal_difference > 0 &&
78  q.normal().squaredNorm() > 0 &&
79  p.normal().squaredNorm() > 0) {
80  const Scalar norm_threshold =
81  0.5 * options.max_normal_difference * M_PI / 180.0;
82  const double first_normal_angle = (q.normal() - p.normal()).norm();
83  const double second_normal_angle = (q.normal() + p.normal()).norm();
84  // Take the smaller normal distance.
85  const Scalar first_norm_distance =
86  std::min(std::abs(first_normal_angle - pair_normals_angle),
87  std::abs(second_normal_angle - pair_normals_angle));
88  // Verify appropriate angle between normals and distance.
89 
90  if (first_norm_distance > norm_threshold) return res;
91  }
92  // Verify restriction on the rotation angle, translation and colors.
93  if (options.max_color_distance > 0) {
94  const bool use_rgb = (p.rgb()[0] >= 0 && q.rgb()[0] >= 0 &&
95  b0.rgb()[0] >= 0 &&
96  b1.rgb()[0] >= 0);
97  bool color_good = (p.rgb() - b0.rgb()).norm() <
98  options.max_color_distance &&
99  (q.rgb() - b1.rgb()).norm() <
100  options.max_color_distance;
101 
102  if (use_rgb && ! color_good) return res;
103  }
104 
105  if (options.max_translation_distance > 0) {
106  const bool dist_good = (p.pos() - b0.pos()).norm() <
107  options.max_translation_distance &&
108  (q.pos() - b1.pos()).norm() <
109  options.max_translation_distance;
110  if (! dist_good) return res;
111  }
112 
113  // need cleaning here
114  if (options.max_angle > 0){
115  VectorType segment2 = (q.pos() - p.pos()).normalized();
116  if (std::acos(segment1.dot(segment2)) <= options.max_angle * M_PI / 180.0) {
117  res.second = true;
118  }
119 
120  if (std::acos(segment1.dot(- segment2)) <= options.max_angle * M_PI / 180.0) {
121  // Add ordered pair.
122  res.first = true;
123  }
124  }else {
125  res.first = true;
126  res.second = true;
127  }
128  return res;
129  }
130  };
131 }
