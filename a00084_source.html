50  bool Match4pcsBase<_Functor, TransformVisitor, PairFilteringFunctor, PFO>::TryQuadrilateral(
51  typename Point3D::Scalar &invariant1,
52  typename Point3D::Scalar &invariant2,
53  int &id1, int &id2, int &id3, int &id4) {
54 
55  Scalar min_distance = std::numeric_limits<Scalar>::max();
56  int best1, best2, best3, best4;
57  best1 = best2 = best3 = best4 = -1;
58  for (int i = 0; i < 4; ++i) {
59  for (int j = 0; j < 4; ++j) {
60  if (i == j) continue;
61  int k = 0;
62  while (k == i || k == j) k++;
63  int l = 0;
64  while (l == i || l == j || l == k) l++;
65  Scalar local_invariant1;
66  Scalar local_invariant2;
67  // Compute the closest points on both segments, the corresponding
68  // invariants and the distance between the closest points.
69  Scalar segment_distance = distSegmentToSegment(
70  MatchBaseType::base_3D_[i].pos(), MatchBaseType::base_3D_[j].pos(),
71  MatchBaseType::base_3D_[k].pos(), MatchBaseType::base_3D_[l].pos(),
72  local_invariant1, local_invariant2);
73  // Retail the smallest distance and the best order so far.
74  if (segment_distance < min_distance) {
75  min_distance = segment_distance;
76  best1 = i;
77  best2 = j;
78  best3 = k;
79  best4 = l;
80  invariant1 = local_invariant1;
81  invariant2 = local_invariant2;
82  }
83  }
84  }
85 
86  if(best1 < 0 || best2 < 0 || best3 < 0 || best4 < 0 ) return false;
87 
88  Coordinates tmp = MatchBaseType::base_3D_;
89  MatchBaseType::base_3D_[0] = tmp[best1];
90  MatchBaseType::base_3D_[1] = tmp[best2];
91  MatchBaseType::base_3D_[2] = tmp[best3];
92  MatchBaseType::base_3D_[3] = tmp[best4];
93 
94  CongruentBaseType tmpId = {id1, id2, id3, id4};
95  id1 = tmpId[best1];
96  id2 = tmpId[best2];
97  id3 = tmpId[best3];
98  id4 = tmpId[best4];
99 
100  return true;
101  }

107  bool Match4pcsBase<_Functor, TransformVisitor, PairFilteringFunctor, PFO>::SelectQuadrilateral(
108  Scalar &invariant1,
109  Scalar &invariant2,
110  int& base1, int& base2, int& base3, int& base4) {
111 
112  const Scalar kBaseTooSmall (0.2);
113  int current_trial = 0;
114 
115  // Try fix number of times.
116  while (current_trial < MatchBaseType::kNumberOfDiameterTrials) {
117  // Select a triangle if possible. otherwise fail.
118  if (!MatchBaseType::SelectRandomTriangle(base1, base2, base3)){
119  return false;
120  }
121 
122  const auto& b0 = MatchBaseType::base_3D_[0] = MatchBaseType::sampled_P_3D_[base1];
123  const auto& b1 = MatchBaseType::base_3D_[1] = MatchBaseType::sampled_P_3D_[base2];
124  const auto& b2 = MatchBaseType::base_3D_[2] = MatchBaseType::sampled_P_3D_[base3];
125 
126  // The 4th point will be a one that is close to be planar to the other 3
127  // while still not too close to them.
128  const double x1 = b0.x();
129  const double y1 = b0.y();
130  const double z1 = b0.z();
131  const double x2 = b1.x();
132  const double y2 = b1.y();
133  const double z2 = b1.z();
134  const double x3 = b2.x();
135  const double y3 = b2.y();
136  const double z3 = b2.z();
137 
138  // Fit a plan.
139  Scalar denom = (-x3 * y2 * z1 + x2 * y3 * z1 + x3 * y1 * z2 - x1 * y3 * z2 -
140  x2 * y1 * z3 + x1 * y2 * z3);
141 
142  if (denom != 0) {
143  Scalar A =
144  (-y2 * z1 + y3 * z1 + y1 * z2 - y3 * z2 - y1 * z3 + y2 * z3) / denom;
145  Scalar B =
146  (x2 * z1 - x3 * z1 - x1 * z2 + x3 * z2 + x1 * z3 - x2 * z3) / denom;
147  Scalar C =
148  (-x2 * y1 + x3 * y1 + x1 * y2 - x3 * y2 - x1 * y3 + x2 * y3) / denom;
149  base4 = -1;
150  Scalar best_distance = std::numeric_limits<Scalar>::max();
151  // Go over all points in P.
152  const Scalar too_small = std::pow(MatchBaseType::max_base_diameter_ * kBaseTooSmall, 2);
153  for (unsigned int i = 0; i < MatchBaseType::sampled_P_3D_.size(); ++i) {
154  const auto &p = MatchBaseType::sampled_P_3D_[i];
155  if ((p.pos() - b0.pos()).squaredNorm() >= too_small &&
156  (p.pos() - b1.pos()).squaredNorm() >= too_small &&
157  (p.pos() - b2.pos()).squaredNorm() >= too_small) {
158  // Not too close to any of the first 3.
159  const Scalar distance =
160  std::abs(A * p.x() + B * p.y() + C * p.z() - 1.0);
161  // Search for the most planar.
162  if (distance < best_distance) {
163  best_distance = distance;
164  base4 = int(i);
165  }
166  }
167  }
168  // If we have a good one we can quit.
169  if (base4 != -1) {
170  MatchBaseType::base_3D_[3] = MatchBaseType::sampled_P_3D_[base4];
171  if(TryQuadrilateral(invariant1, invariant2, base1, base2, base3, base4))
172  return true;
173  }
174  }
175  current_trial++;
176  }
177 
178  // We failed to find good enough base..
179  return false;
180  }

278  typename Match4pcsBase<_Functor, TransformVisitor, PairFilteringFunctor, PFO>::Scalar
279  Match4pcsBase<_Functor, TransformVisitor, PairFilteringFunctor, PFO>::distSegmentToSegment(
280  const VectorType& p1, const VectorType& p2,
281  const VectorType& q1, const VectorType& q2,
282  Scalar& invariant1, Scalar& invariant2) {
283 
284  static const Scalar kSmallNumber = 0.0001;
285  VectorType u = p2 - p1;
286  VectorType v = q2 - q1;
287  VectorType w = p1 - q1;
288  Scalar a = u.dot(u);
289  Scalar b = u.dot(v);
290  Scalar c = v.dot(v);
291  Scalar d = u.dot(w);
292  Scalar e = v.dot(w);
293  Scalar f = a * c - b * b;
294  // s1,s2 and t1,t2 are the parametric representation of the intersection.
295  // they will be the invariants at the end of this simple computation.
296  Scalar s1 = 0.0;
297  Scalar s2 = f;
298  Scalar t1 = 0.0;
299  Scalar t2 = f;
300 
301  if (f < kSmallNumber) {
302  s1 = 0.0;
303  s2 = 1.0;
304  t1 = e;
305  t2 = c;
306  } else {
307  s1 = (b * e - c * d);
308  t1 = (a * e - b * d);
309  if (s1 < 0.0) {
310  s1 = 0.0;
311  t1 = e;
312  t2 = c;
313  } else if (s1 > s2) {
314  s1 = s2;
315  t1 = e + b;
316  t2 = c;
317  }
318  }
319 
320  if (t1 < 0.0) {
321  t1 = 0.0;
322  if (-d < 0.0)
323  s1 = 0.0;
324  else if (-d > a)
325  s1 = s2;
326  else {
327  s1 = -d;
328  s2 = a;
329  }
330  } else if (t1 > t2) {
331  t1 = t2;
332  if ((-d + b) < 0.0)
333  s1 = 0;
334  else if ((-d + b) > a)
335  s1 = s2;
336  else {
337  s1 = (-d + b);
338  s2 = a;
339  }
340  }
341  invariant1 = (std::abs(s1) < kSmallNumber ? 0.0 : s1 / s2);
342  invariant2 = (std::abs(t1) < kSmallNumber ? 0.0 : t1 / t2);
343 
344  return ( w + (invariant1 * u) - (invariant2 * v)).norm();
345  }
346 }
